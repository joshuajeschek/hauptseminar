\defbibfilter{articles}{
  type=article or
  type=inproceedings or
  type=report
}

\defbibfilter{books}{
  type=book or
  type=inbook or
  type=incollection
}

\defbibfilter{software}{
  type=software
}

\defbibfilter{other}{
  not (
    type=article or
    type=inproceedings or
    type=report or
    type=book or
    type=inbook or
    type=incollection or
    type=software
  )
}
